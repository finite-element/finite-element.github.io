[docs]def lagrange_points(cell, degree):
    """Construct the locations of the equispaced Lagrange nodes on cell.

    :param cell: the :class:`~.reference_elements.ReferenceCell`
    :param degree: the degree of polynomials for which to construct nodes.

    :returns: a rank 2 :class:`~numpy.array` whose rows are the
        coordinates of the nodes.

    The implementation of this function is left as an :ref:`exercise
    <ex-lagrange-points>`.

    """
    if cell is ReferenceInterval:

        # x_i = a + (b - a) * (i / k) 
        # a = 0, b = 1 for ReferenceInterval 
        # k = degree => x_i = i / degree
        return np.array([[i / degree] for i in range(degree + 1)])
        
    elif cell is ReferenceTriangle:
        
        # x_{i,j} = (i / degree, j / degree) for 0 <= i + j <= degree
        # swap i and j for lexicographical ordering of vertices
        return np.array([[j / degree, i / degree] for i in range(degree + 1) for j in range(degree - i + 1)])

    else:
        raise ValueError("Unknown reference cell")



[docs]def vandermonde_matrix(cell, degree, points, grad=False):
    """Construct the generalised Vandermonde matrix for polynomials of the
    specified degree on the cell provided.

    :param cell: the :class:`~.reference_elements.ReferenceCell`
    :param degree: the degree of polynomials for which to construct the matrix.
    :param points: a list of coordinate tuples corresponding to the points.
    :param grad: whether to evaluate the Vandermonde matrix or its gradient.

    :returns: the generalised :ref:`Vandermonde matrix <sec-vandermonde>`

    The implementation of this function is left as an :ref:`exercise
    <ex-vandermonde>`.
    """

    # Ensure points are an array
    points = np.array(points)

    if cell is ReferenceInterval:

        # number of points
        m = len(points)

        if grad:
            # initialise tensor
            v_g = np.empty(shape=(m, degree + 1, 1))

            # derivative of constant is 0
            v_g[:, 0] = [0]
            for i in range(1, degree + 1):

                # derivative of (x_j)^i = i(x_j)^{i - 1}
                derv = i * np.power(points, i - 1)
                v_g[:, i] = derv

            return v_g

        else:
            # initialise matrix
            v = np.empty(shape=(m, degree + 1))

            for i in range(degree + 1):
                v[:, i] = np.power(points, i)[:, 0]

            return v 

    elif cell is ReferenceTriangle:

        # number of points 
        m = len(points)

        # number of nodes is (degree + dimension) C (dimension)
        num_nodes = int(comb(degree + cell.dim, cell.dim))

        if grad:
            # initialise tensor
            v_g = np.empty(shape=(m, num_nodes, 2))

            # 'unzip' points into x and y coordinates
            vec_x = points[:, 0]
            vec_y = points[:, 1]

            c = 0
            for i in range(degree + 1):
                # simultaneously count down and up for power of x and y to get binomial expansion
                for pow in range(i + 1):
                    # grad (x^a)(y^b) = [(ax^{a-1})(y^b), (x^a)(by^{b-1})]

                    # partial x component
                    if i - pow - 1 < 0:
                        x_contribution_x = 0
                    else:
                        x_contribution_x = (i - pow) * np.power(vec_x, i - pow - 1)

                    y_contribution_x = np.power(vec_y, pow)
                    partial_x = np.multiply(x_contribution_x, y_contribution_x)

                    # partial y component
                    x_contribution_y = np.power(vec_x, i - pow)
                    if pow - 1 < 0:
                        y_contribution_y = 0
                    else:
                        y_contribution_y = pow * np.power(vec_y, pow - 1)
                    partial_y = np.multiply(x_contribution_y, y_contribution_y)

                    v_g[:, c] = np.dstack((partial_x, partial_y))
                    c += 1

            return v_g

        else:
            # initialise matrix
            v = np.empty(shape=(m, num_nodes))

            # 'unzip' points into x and y coordinates
            vec_x = points[:, 0]
            vec_y = points[:, 1]

            c = 0
            for i in range(degree + 1):
                # simultaneously count down and up for power of x and y to get binomial expansion
                for pow in range(i + 1):
                    x_contribution = np.power(vec_x, i - pow)
                    y_contribution = np.power(vec_y, pow)
                    v[:, c] = np.multiply(x_contribution, y_contribution)
                    c += 1

            return v 

    else:
        raise ValueError("Unknown reference cell")







[docs]class FiniteElement(object):
    def __init__(self, cell, degree, nodes, entity_nodes=None):
        """A finite element defined over cell.

        :param cell: the :class:`~.reference_elements.ReferenceCell`
            over which the element is defined.
        :param degree: the
            polynomial degree of the element. We assume the element
            spans the complete polynomial space.
        :param nodes: a list of coordinate tuples corresponding to
            point evaluation node locations on the element.
        :param entity_nodes: a dictionary of dictionaries such that
            entity_nodes[d][i] is the list of nodes associated with
            entity `(d, i)` of dimension `d` and index `i`.

        Most of the implementation of this class is left as exercises.
        """

        #: The :class:`~.reference_elements.ReferenceCell`
        #: over which the element is defined.
        self.cell = cell
        #: The polynomial degree of the element. We assume the element
        #: spans the complete polynomial space.
        self.degree = degree
        #: The list of coordinate tuples corresponding to the nodes of
        #: the element.
        self.nodes = nodes
        #: A dictionary of dictionaries such that ``entity_nodes[d][i]``
        #: is the list of nodes associated with entity `(d, i)`.
        self.entity_nodes = entity_nodes

        if entity_nodes:
            #: ``nodes_per_entity[d]`` is the number of entities
            #: associated with an entity of dimension d.
            self.nodes_per_entity = np.array([len(entity_nodes[d][0])
                                              for d in range(cell.dim+1)])

        # Replace this exception with some code which sets
        # self.basis_coefs
        # to an array of polynomial coefficients defining the basis functions.
        v = vandermonde_matrix(cell=cell, degree=degree, points=nodes)
        self.basis_coefs = np.linalg.inv(v)

        #: The number of nodes in this element.
        self.node_count = nodes.shape[0]


[docs]    def tabulate(self, points, grad=False):
        """Evaluate the basis functions of this finite element at the points
        provided.

        :param points: a list of coordinate tuples at which to
            tabulate the basis.
        :param grad: whether to return the tabulation of the basis or the
            tabulation of the gradient of the basis.

        :result: an array containing the value of each basis function
            at each point. If `grad` is `True`, the gradient vector of
            each basis vector at each point is returned as a rank 3
            array. The shape of the array is (points, nodes) if
            ``grad`` is ``False`` and (points, nodes, dim) if ``grad``
            is ``True``.

        The implementation of this method is left as an :ref:`exercise
        <ex-tabulate>`.

        """
        v = vandermonde_matrix(cell=self.cell, degree=self.degree, points=points, grad=grad)

        if grad:
            # grad phi_j(X_i)e_k = sum_b (grad(V))_{ibk}C_{bj}
            return np.einsum("ibk,bj->ijk", v, self.basis_coefs)
        else:
            # T = VC
            return v @ self.basis_coefs




[docs]    def interpolate(self, fn):
        """Interpolate fn onto this finite element by evaluating it
        at each of the nodes.

        :param fn: A function ``fn(X)`` which takes a coordinate
           vector and returns a scalar value.

        :returns: A vector containing the value of ``fn`` at each node
           of this element.

        The implementation of this method is left as an :ref:`exercise
        <ex-interpolate>`.

        """

        return [fn(x) for x in self.nodes]



    def __repr__(self):
        return "%s(%s, %s)" % (self.__class__.__name__,
                               self.cell,
                               self.degree)




[docs]class LagrangeElement(FiniteElement):
    def __init__(self, cell, degree):
        """An equispaced Lagrange finite element.

        :param cell: the :class:`~.reference_elements.ReferenceCell`
            over which the element is defined.
        :param degree: the
            polynomial degree of the element. We assume the element
            spans the complete polynomial space.

        The implementation of this class is left as an :ref:`exercise
        <ex-lagrange-element>`.
        """

        # Use lagrange_points to obtain the set of nodes.  Once you
        # have obtained nodes, the following line will call the
        # __init__ method on the FiniteElement class to set up the
        # basis coefficients.

        nodes = lagrange_points(cell=cell, degree=degree)

        # Instantiate entity node
        entity_node = {}
        for (dim, entities) in cell.topology.items():
            entity_node[dim] = {}
            for index in entities.keys():
                entity_node[dim][index] = []

        def find_associated_entity(i, node):
            # Iterate through topology
            for (dim, entities) in cell.topology.items():
                # Iterate through entity instances
                for index in entities.keys():
                    if cell.point_in_entity(node, (dim, index)):
                        entity_node[dim][index].append(i)
                        return # To avoid duplicating nodes part of several entities

        # Iterate through nodes in order defined in lagrange points
        for i, node in enumerate(nodes):
            find_associated_entity(i, node)

        super(LagrangeElement, self).__init__(cell, degree, nodes, entity_nodes=entity_node)



[docs]class VectorFiniteElement(object):

    def __init__(self, scalar_element):
        """A vector finite element defined over cell.

        :param scalar_element: the :class:`~.finite_elements.FiniteElement`
            from which the vector finite element is extended.
        """

        #: Reference to scalar finite element this vector finite element was extended from
        self.scalar_element = scalar_element
        #: The :class:`~.reference_elements.ReferenceCell`
        #: over which the element is defined.
        self.cell = scalar_element.cell
        #: Vector dimension (assumed to be equal to element geometric and topologcal dimension)
        self.dim = scalar_element.cell.dim 
        #: The polynomial degree of the element. We assume the element
        #: spans the complete polynomial space.
        self.degree = scalar_element.degree

        # Extend entity nodes of finite element provided
        scalar_en = scalar_element.entity_nodes
        for (dim, entities) in scalar_en.items():
            for (entity, nodes) in entities.items():
                scalar_en[dim][entity] = list(chain.from_iterable((2*n, 2*n + 1) for n in nodes))

        #: A dictionary of dictionaries such that ``entity_nodes[d][i]``
        #: is the list of nodes associated with entity `(d, i)`.
        self.entity_nodes = scalar_en
        
        #: ``nodes_per_entity[d]`` is the number of entities
        #: associated with an entity of dimension d.
        # dimension d times more nodes per entity
        self.nodes_per_entity = scalar_element.nodes_per_entity * self.dim 

        #: A list of coordinate tuples corresponding to point evaluation node 
        #: locations on the element.
        # Same as list of nodal points from scalar element with each point
        # repeated dimension d times
        self.nodes = list(chain.from_iterable((n,) * self.dim for n in scalar_element.nodes))

        #: The corresponding canonical basis vector to contract with the 
        #: the elements nodes defined above.
        basis_vectors = [[0 if i != e else 1 for i in range(self.dim)] for e in range(self.dim)]
        self.node_weights = np.array(basis_vectors*scalar_element.node_count)

        #: The number of nodes in this element.
        self.node_count = len(self.nodes)



[docs]    def tabulate(self, points, grad=False):
        """Evaluate the basis functions of this vector finite element at the 
        points provided.

        :param points: a list of coordinate tuples at which to
            tabulate the basis.
        :param grad: whether to return the tabulation of the basis or the
            tabulation of the gradient of the basis.

        :result: an array containing the value of each basis function
            at each point. If `grad` is `True`, the gradient vector of
            each basis vector at each point is returned as a rank 4
            array. The shape of the array is (points, nodes, dim) if
            ``grad`` is ``False`` and (points, nodes, dim, dim) if ``grad``
            is ``True``.

        """
        
        # Tabulate over scalar version of element
        scalar_tab = self.scalar_element.tabulate(points, grad)

        # Create extended tabulate array
        num_points = scalar_tab.shape[0]
        # 'dimension' times as many nodes
        num_nodes = self.dim * scalar_tab.shape[1]
        # If grad True then extra dimension 
        if grad:
            num_dir = scalar_tab.shape[2]
            vector_tab = np.zeros(shape=(num_points, num_nodes, self.dim, num_dir))
        else:
            vector_tab = np.zeros(shape=(num_points, num_nodes, self.dim))

        # Iterate and apply conversion formula
        for i in range(num_points):
            for j in range(num_nodes):
                # Check if node is pointing in x or y direction
                direction = j % self.dim
                if grad:
                    # Node in x direction gets corresponding gradient in x direction & v.v
                    for k in range(self.dim):
                        vector_tab[i, j, k, direction] = scalar_tab[i, j//self.dim, k]
                else:
                    vector_tab[i, j, direction] = scalar_tab[i, j//self.dim]

        return vector_tab
