[docs]def assemble(fs, f):
    """Assemble the finite element system for the Stokes problem given
    the function spaces in which to solve and the right hand side
    function."""

    # Separate W = V X Q
    fs_v, fs_q = fs[0], fs[1]
    f_v = f[0]
    element_v, element_q = fs_v.element, fs_q.element
    mesh_v, mesh_q = fs_v.mesh, fs_q.mesh
    cell_nodes_v, cell_nodes_q = fs_v.cell_nodes, fs_q.cell_nodes

    # Create an appropriate (complete) quadrature rule
    degree = 2 * max(element_v.degree , element_q.degree)
    Q_v = gauss_quadrature(element_v.cell, degree)
    Q_q = gauss_quadrature(element_q.cell, degree)

    # Tabulate the basis functions and their gradients at the quadrature points
    phi, psi = element_v.tabulate(Q_v.points), element_q.tabulate(Q_q.points)
    g_phi = element_v.tabulate(Q_v.points, grad=True)

	# Left hand side matrix blocks
    # A is |V| x |V| matrix
    A = sp.lil_matrix((fs_v.node_count, fs_v.node_count))
    # B is |Q| x |V| matrix
    B = sp.lil_matrix((fs_q.node_count, fs_v.node_count))

    # Right hand side vector
    # l is |V| + |Q| vector
    l = np.zeros(fs_v.node_count + fs_q.node_count)

    # See mastery.ipynb for derivation
    def f_contribution(c):

        # f_{M(c, j)}
        nodes = cell_nodes_v[c]
        node_coeffs = np.array([f_v.values[x] for x in nodes])

		# \sum_j f_{M(c, j)} W_j
        # phi has indices [point, node, direction] = [q, k, x]
        f_cont = np.einsum("j,qjx->qx", node_coeffs, phi)

        # dot with i^th local basis function of V and quadrature weight
        return np.einsum("qix,qx,q->i", phi, f_cont, Q_v.weights)

    # See mastery.ipynb for derivation
    def a_contribution(J):

        inv_J = np.linalg.inv(J)

        # g_phi has indices [point, node, dim, direction] = [q, k, d, x]
        e_phi_j = np.einsum("yb,qjya->qjab", inv_J, g_phi) + np.einsum("ya,qjyb->qjab", inv_J, g_phi)
        e_phi_i = np.einsum("lb,qila->qiab", inv_J, g_phi) + np.einsum("la,qilb->qiab", inv_J, g_phi)

        # sum over quadrature weights as well
        return np.einsum("qjab,qiab,q->ij", e_phi_j, e_phi_i, Q_v.weights)

    # Loop over all the cells in space V and assemble A and l
    print("assembling A & l")
    for c in range(mesh_v.entity_counts[-1]):

        # Global numbering of cell's nodes
        nodes = cell_nodes_v[c]

        # |J_W| = |J_V||J_Q|
        J_V = mesh_v.jacobian(c)
        J_Q = mesh_q.jacobian(c)
        detJ_V = np.abs(np.linalg.det(J_V))
        detJ_Q = np.abs(np.linalg.det(J_Q))
        detJ_W = detJ_V * detJ_Q

        # update l (this leaves fs_q untouched)
        l[nodes] += f_contribution(c) * detJ_W

        # update A
        A[np.ix_(nodes, nodes)] += a_contribution(J_V) * detJ_W * 0.25

    # See mastery.ipynb for derivation
    def b_contribution(J):

        inv_J = np.linalg.inv(J)

        # div \phi_i
        # g_phi has indices [point, node, dim, direction] = [q, i, a, b]
        div_phi_i = np.einsum("ba,qiab->qi", inv_J, g_phi)

        # psi has indices [point, node] = [q, i]
        # sum over quadrature weights as well
        return np.einsum("qi,qj,q->ij", psi, div_phi_i, Q_q.weights)
    
    # Loop over "cells in space W" (just V and Q simultaneosly) and assemble B
    print("assembling B")
    # V & Q from same mesh so same cell counts/ordering
    for c in range(mesh_v.entity_counts[-1]):

        # global numbering of each cell's nodes
        nodes_v = cell_nodes_v[c]
        nodes_q = cell_nodes_q[c]

        # |J_W| = |J_V||J_Q|
        J_V = mesh_v.jacobian(c)
        J_Q = mesh_q.jacobian(c)
        detJ_W = np.abs(np.linalg.det(J_V))* np.abs(np.linalg.det(J_Q))

        # update B
        B[np.ix_(nodes_q, nodes_v)] += b_contribution(J_V) * detJ_W

    print("Constructing block matrix")
    # Combine A and B into block matrix
    C = sp.bmat([[A, B.T], [B, None]], "lil")

	# Boundary conditions
    print("Imposing boundary conditions")

    # Vanish at boundary
    bn_v = boundary_nodes(fs_v, vector_space=True)
    for bn in bn_v:
        C[bn] = 0
        l[bn] = 0
        C[bn, bn] = 1

    # L2^o - we arbitrarily select basis function \psi_0 in Q
    # Account for A block (l[m] already 0)
    m = fs_v.node_count
    C[m] = 0
    C[m, m] = 1

    return C, l



[docs]def boundary_nodes(fs, vector_space=False):
    """Find the list of boundary nodes in fs. This is a
    unit-square-specific solution. A more elegant solution would employ
    the mesh topology and numbering.
    """
    eps = 1.e-10

    f = Function(fs)

    def on_boundary(x):
        """Return 1 if on the boundary, 0. otherwise."""
        if x[0] < eps or x[0] > 1 - eps or x[1] < eps or x[1] > 1 - eps:
            return [1., 1.] if vector_space else 1.
        else:
            return [0., 0.] if vector_space else 0.

    f.interpolate(on_boundary)

    return np.flatnonzero(f.values)




[docs]def solve_mastery(resolution, analytic=False, return_error=False):
    """This function should solve the mastery problem with the given
    resolution. It should return both the solution
    :class:`~fe_utils.function_spaces.Function` and the :math:`L^2` error in
    the solution.

    If ``analytic`` is ``True`` then it should not solve the equation
    but instead return the analytic solution. If ``return_error`` is
    true then the difference between the analytic solution and the
    numerical solution should be returned in place of the solution.
    """

    # Square mesh
    mesh = UnitSquareMesh(resolution, resolution)

    # Function space for V
    # (P2)^2: Quadratic lagrange for each cartesian component
    scalar_fe_v = LagrangeElement(mesh.cell, degree=2)
    vector_fe_v = VectorFiniteElement(scalar_fe_v)
    fs_v = FunctionSpace(mesh, vector_fe_v)

    # Function space for Q
    # (P1): Linear lagrange scalar element
    fe_q = LagrangeElement(mesh.cell, degree=1)
    fs_q = FunctionSpace(mesh, fe_q)

    # Create a function to hold the analytic solution for comparison purposes.
    # See mastery.ipynb for derivation
    print("Interpolating analytic function")
    aa_u = Function(fs_v)
    aa_u.interpolate(lambda x: (2*pi*(cos(2*pi*x[0])-1)*sin(2*pi*x[1]), 2*pi*(1-cos(2*pi*x[1]))*sin(2*pi*x[0])))

	# Constant pressure = 0
    aa_p = Function(fs_q)
    aa_p.interpolate(lambda _: 0)

    # If the analytic answer has been requested then bail out now.
    if analytic:
        return (aa_u, aa_p), 0.0

    # Create the right hand side function 
    # See mastery.ipynb for derivation
    # We split f into the two function spaces V and Q
    print("Manufacturing forcing function")
    f_v = Function(fs_v)
    f_q = Function(fs_q)
    # Using solution manufacturing, we plug in u as above and p = 0
    def f_x(x):
        return - 4 * pi ** 3 * (1 + 2 * cos(2 * pi * x[0])) * sin(2 * pi * x[1])
    def f_y(x):
        return 4 * pi ** 3 * (2 * cos(2 * pi * x[1]) - 1) * sin(2 * pi * x[0])
    f_v.interpolate(lambda x: [f_x(x), f_y(x)])
    f_q.interpolate(lambda _ : 0)

    # See mastery.ipynb for derivation
    # Assemble the finite element system.
    print("Assembling finite element system")
    A, l = assemble((fs_v, fs_q), (f_v, f_q))

    # Create the functions to hold the solutions
    u = Function(fs_v)
    p = Function(fs_q)

    print("Solving finite element system")
    # Cast the matrix to a sparse format
    A = sp.csr_matrix(A)
    # LU decomposition of matrix to use SuperLU solver
    LU = splinalg.splu(A)
    w = LU.solve(l)
    # Split coefficients
    m = fs_v.node_count
    u.values = w[:m]
    p.values = w[m:]

    print("Computing L2 error")
    # Compute the L^2 error in the solution for testing purposes.
    error_u = errornorm(u, aa_u)
    error_p = errornorm(p, aa_p)
    error = (error_u ** 2 + error_p ** 2) ** 0.5
    print(f'error: {error}, of which p: {error_p}, of which u {error_u}')

    if return_error:
        u.values -= aa_u.values
        p.values -= aa_p.values

    # Return the solution and the error in the solution.
    return (u, p), error
